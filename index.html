Kết nối Google Sheets

Để kết nối với Google Sheets, bạn cần:
1. Mở Google Sheets → Extensions → Apps Script
2. Dán đoạn code Apps Script (xem bên dưới) → Deploy → Web App
3. Sao chép URL Web App và dán vào ô bên dưới
Vào Google Sheets → Extensions → Apps Script → Deploy → Copy URL
// Apps Script: dán vào script.google.com
function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName("DIEM_DEN");
  var data = sh.getDataRange().getValues();
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
function doPost(e) {
  var p = JSON.parse(e.postData.contents);
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName(p.sheet || "DIEM_DEN");
  if(p.action === "update") {
    sh.getRange(p.row,p.col).setValue(p.value);
  } else if(p.action === "append") {
    sh.appendRow(p.row);
  }
  return ContentService
    .createTextOutput("OK")
    .setMimeType(ContentService.MimeType.TEXT);
}
